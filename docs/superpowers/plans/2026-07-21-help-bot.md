# MediPrimer Help Bot Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a RAG-backed chat widget on mediprimer.org that answers Medicare/Medicaid questions from a local knowledge base (the site's own pages + curated official-source docs), falling back to general knowledge with a caveat when the KB doesn't cover the question — never recommending a specific plan or carrier.

**Architecture:** A gitignored `kb/` directory holds a small Python toolchain: chunk + embed MediPrimer's pages and a curated official-doc list into a local SQLite vector store (`sentence-transformers`, CPU, no new paid vendor), retrieve top-K chunks by cosine similarity at query time, and generate answers with the Claude API. A FastAPI service (`kb/server.py`) exposes this as `POST /api/chat`, reverse-proxied and rate-limited by nginx, running as a systemd unit. The frontend is a vanilla-JS widget (`public/chatbot.js`) injected on every page by the existing `build/normalize.py`.

**Tech Stack:** Python 3.12, `sentence-transformers` (local CPU embeddings, model `all-MiniLM-L6-v2`), SQLite, FastAPI + uvicorn, `anthropic` SDK (model `claude-haiku-4-5-20251001` — cheap/fast, appropriate for a rate-limited public endpoint), vanilla JS (no framework, matching the rest of `public/`), nginx, systemd.

## Global Constraints

- Never recommend a specific plan, carrier, or "you should pick X" (spec: Goals).
- Answer from the knowledge base first; if uncovered, answer from general knowledge but say so explicitly and add a "verify at [official source]" caveat (spec: Backend service).
- Every factual claim must carry a source link — MediPrimer page and/or official source (spec: Knowledge base, Backend service).
- No server-side conversation storage. Each `/api/chat` request is self-contained (spec: Non-goals, Backend service).
- No new paid vendor dependency (spec: Goals) — embeddings run locally, not via a paid embeddings API.
- The site must remain fully usable with JavaScript disabled; the widget must degrade gracefully (spec: Frontend widget).
- `kb/` is gitignored, same treatment as `seo/` and `update/` (spec: Knowledge base).
- Official-doc URL list requires Kurt's sign-off before anything is fetched (spec: Open questions #1) — Task 4 stops for this explicitly.

# kb/chunker.py
"""Turn a MediPrimer (or official-source) HTML page into retrieval chunks.
No new dependency: same regex-based tag stripping style as
seo/analyze.py's visible_text(), reused here for consistency."""
import html
import re

DROP_BLOCKS = re.compile(
    r"<script.*?</script>|<style.*?</style>|"
    r"<header[^>]*>.*?</header>|<footer[^>]*>.*?</footer>|"
    r"<nav[^>]*>.*?</nav>",
    re.DOTALL | re.IGNORECASE,
)
HEADING_SPLIT = re.compile(r"(<h[1-3][^>]*>.*?</h[1-3]>)", re.DOTALL | re.IGNORECASE)
TAG_RE = re.compile(r"<[^>]+>")


def _visible_text(fragment_html):
    text = TAG_RE.sub(" ", fragment_html)
    text = html.unescape(text)
    return re.sub(r"\s+", " ", text).strip()


def chunk_html(page_html, source_url):
    """Return [{"text": ..., "source_url": ...}, ...] for one page.
    Splits on h1/h2/h3 boundaries so each chunk stays topically narrow —
    that's what keeps retrieval precise at query time."""
    body = DROP_BLOCKS.sub(" ", page_html)
    parts = HEADING_SPLIT.split(body)

    chunks = []
    current = ""
    for part in parts:
        if HEADING_SPLIT.match(part):
            if current.strip():
                text = _visible_text(current)
                if text:
                    chunks.append({"text": text, "source_url": source_url})
            current = part
        else:
            current += part
    if current.strip():
        text = _visible_text(current)
        if text:
            chunks.append({"text": text, "source_url": source_url})
    return chunks
```

# kb/store.py
"""SQLite-backed chunk + embedding store. Embeddings are stored as raw
float32 bytes — at this corpus size (hundreds of chunks) a linear scan
with numpy cosine similarity at query time is fast enough that no
vector-index library is needed."""
import sqlite3
import numpy as np


class Store:
    def __init__(self, db_path):
        self.db_path = db_path

    def _conn(self):
        return sqlite3.connect(self.db_path)

    def reset(self):
        with self._conn() as c:
            c.execute("DROP TABLE IF EXISTS chunks")
            c.execute(
                """CREATE TABLE chunks (
                    id INTEGER PRIMARY KEY,
                    text TEXT NOT NULL,
                    source_url TEXT NOT NULL,
                    embedding BLOB NOT NULL
                )"""
            )

    def add(self, chunks, embeddings):
        assert len(chunks) == len(embeddings), "chunks/embeddings length mismatch"
        with self._conn() as c:
            for chunk, emb in zip(chunks, embeddings):
                c.execute(
                    "INSERT INTO chunks (text, source_url, embedding) VALUES (?, ?, ?)",
                    (chunk["text"], chunk["source_url"],
                     np.asarray(emb, dtype="float32").tobytes()),
                )

    def all(self):
        with self._conn() as c:
            rows = c.execute("SELECT text, source_url, embedding FROM chunks").fetchall()
        return [
            {"text": text, "source_url": source_url,
             "embedding": np.frombuffer(blob, dtype="float32")}
            for text, source_url, blob in rows
        ]
```

"""Rebuild kb/index.sqlite from public/*.html. Run manually or via the
weekly refresh (Task 4/5)."""
import glob
import pathlib
import sys

sys.path.insert(0, str(pathlib.Path(__file__).parent))
from chunker import chunk_html
from embedder import embed
from store import Store

HERE = pathlib.Path(__file__).parent
PUB = HERE.parent / "public"
DB_PATH = HERE / "index.sqlite"


def mediprimer_chunks():
    chunks = []
    for path in sorted(glob.glob(str(PUB / "*.html"))):
        name = pathlib.Path(path).name
        url = f"https://mediprimer.org/{'' if name == 'index.html' else name}"
        page_html = pathlib.Path(path).read_text(encoding="utf-8")
        chunks.extend(chunk_html(page_html, url))
    return chunks


def main():
    chunks = mediprimer_chunks()
    if not chunks:
        print("build_index: FATAL no chunks produced from public/*.html", file=sys.stderr)
        sys.exit(1)
    embeddings = embed([c["text"] for c in chunks])
    store = Store(str(DB_PATH))
    store.reset()
    store.add(chunks, embeddings)
    print(f"build_index: indexed {len(chunks)} chunks -> {DB_PATH.name}")


if __name__ == "__main__":
    main()

# kb/retrieve.py
import numpy as np

from embedder import embed
from store import Store


def _cosine_topk(query_vec, candidate_matrix, k):
    """Both query_vec and candidate rows are assumed pre-normalized
    (embedder.embed uses normalize_embeddings=True), so dot product
    == cosine similarity. Returns row indices, best-first."""
    scores = candidate_matrix @ query_vec
    order = np.argsort(-scores)
    return order[:k]


def retrieve(question, db_path, k=5):
    store = Store(db_path)
    rows = store.all()
    if not rows:
        return []
    query_vec = embed([question])[0]
    matrix = np.stack([r["embedding"] for r in rows])
    top_idx = _cosine_topk(query_vec, matrix, k)
    scores = matrix @ query_vec
    return [
        {"text": rows[i]["text"], "source_url": rows[i]["source_url"],
         "score": float(scores[i])}
        for i in top_idx
    ]
```

def fetch_all(sources_path):

# kb/server.py
import os
import pathlib

import anthropic
from fastapi import FastAPI
from pydantic import BaseModel, Field

from retrieve import retrieve

HERE = pathlib.Path(__file__).parent
DB_PATH = str(HERE / "index.sqlite")
MODEL = "claude-haiku-4-5-20251001"

SYSTEM_PROMPT = """You are the MediPrimer help bot, answering visitor
questions about Medicare, Medicaid, Medicare Advantage, the ACA
Marketplace, and CHIP.

Hard rules:
1. Answer from the provided knowledge-base context first.
2. If the context doesn't cover the question, say so explicitly, then
   answer from your general knowledge, and add a caveat telling the
   visitor to verify at the relevant official source (medicare.gov,
   ssa.gov, cms.gov, or medicaid.gov).
3. Never recommend a specific plan, carrier, or say "you should pick X."
   You may explain how programs and plan types differ; you may not tell
   someone which one to choose.
4. Cite your sources for factual claims — mention the MediPrimer page
   and/or the official source the fact came from.
5. Keep answers in plain language, short sentences, no jargon left
   unexplained.
"""

app = FastAPI()
anthropic_client = anthropic.Anthropic()  # reads ANTHROPIC_API_KEY from env


class ChatRequest(BaseModel):
    question: str = Field(min_length=1)
    history: list[dict] = Field(default_factory=list)


@app.post("/api/chat")
def chat(req: ChatRequest):
    chunks = retrieve(req.question, DB_PATH, k=5)
    context = "\n\n".join(f"[Source: {c['source_url']}]\n{c['text']}" for c in chunks)
    user_content = (
        f"Knowledge base context:\n{context}\n\n"
        f"Visitor question: {req.question}"
    )
    messages = req.history + [{"role": "user", "content": user_content}]
    response = anthropic_client.messages.create(
        model=MODEL,
        max_tokens=1024,
        system=SYSTEM_PROMPT,
        messages=messages,
    )
    answer = response.content[0].text
    sources = sorted({c["source_url"] for c in chunks})
    return {"answer": answer, "sources": sources}

# kb/refresh.py
"""Weekly KB refresh: re-fetch official docs, re-embed everything, and
flag (never silently apply) any dollar figure that changed since the
last snapshot — same failure mode as the getting-help.html bug fixed in
PR #15, guarded against here at the source instead."""
import re
import sys
from collections import defaultdict

from build_index import mediprimer_chunks, official_chunks, DB_PATH
from embedder import embed
from store import Store

DOLLAR_RE = re.compile(r"\$[0-9]{1,3}(?:,[0-9]{3})+")


def diff_dollar_figures(old_chunks, new_chunks):
    def figures_by_url(chunks):
        by_url = defaultdict(set)
        for c in chunks:
            by_url[c["source_url"]].update(DOLLAR_RE.findall(c["text"]))
        return by_url

    old_by_url = figures_by_url(old_chunks)
    new_by_url = figures_by_url(new_chunks)

    diffs = []
    for url in old_by_url.keys() & new_by_url.keys():
        removed = old_by_url[url] - new_by_url[url]
        added = new_by_url[url] - old_by_url[url]
        if removed or added:
            diffs.append(
                f"{url}: figures changed. Was {sorted(removed)}, now {sorted(added)}"
            )
    return diffs


def main():
    store = Store(str(DB_PATH))
    old_chunks = store.all()  # each row already has text + source_url

    new_chunks = mediprimer_chunks() + official_chunks()
    diffs = diff_dollar_figures(old_chunks, new_chunks)

    if diffs:
        print("refresh: FIGURE CHANGES DETECTED (review before trusting new answers):")
        for d in diffs:
            print(f"  - {d}")
    else:
        print("refresh: no dollar-figure changes detected")

    embeddings = embed([c["text"] for c in new_chunks])
    store.reset()
    store.add(new_chunks, embeddings)
    print(f"refresh: re-indexed {len(new_chunks)} chunks")


if __name__ == "__main__":
    main()
```
